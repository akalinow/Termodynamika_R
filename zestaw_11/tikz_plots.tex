  \coordinate (D) at (2.0, 2.4);   % koniec izotermy T2, poczatek adiabaty do A

  \draw[thick, red, dashed, postaction={decorate}, decoration={markings, mark=at position 0.5 with {\arrow{Latex[length=3mm]}}}]

\def\TypeCViolate{CViolate}%

	% Nodes
	\node[draw, rounded corners, fill=red!15, minimum width=2.0cm, minimum height=1cm] (heater) at (0,0) {$T_H$};
	\node[draw, circle, fill=gray!15, minimum width=1.5cm, minimum height=1cm] (engine) at (0,-2) {$S$};
	\node[draw, rounded corners, fill=blue!15, minimum width=2.0cm, minimum height=1cm] (cooler) at (0,-4) {$T_C$};

    \ifx\EngineType\TypeCarnot
        \draw[->, red!70!black] (heater.south) -- node[right] {$Q_H$} (engine.north);

        \draw[->] (engine.east) -- ++(1,0) node[midway, above] {$W$};
    \fi

    \fi
    %%%%%%%%%%%%%%%%%

        \draw[->, blue!70!black] (cooler.north) -- node[right] {$Q_C$} (engine.south);
    \fi


\end{tikzpicture}%
